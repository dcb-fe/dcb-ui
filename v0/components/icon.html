<template>
  <ul class="list">
    <li
      v-for="item in icons"
      :key="item"
      class="item"
      @click="handleClick(item)"
    >
      <d-icon :name="item" class="icon" />
      <div class="name">{{ item }}</div>
    </li>
  </ul>
</template>

<script>
  import { copyTextToClipboard } from 'vtils';
  import { Icon } from '@dcbfe/ui';

  export default {
    data() {
      return {
        icons: Icon.props.name.enum,
      };
    },
    methods: {
      handleClick(name) {
        copyTextToClipboard(`<d-icon name="${name}" />`);
        alert('已复制');
      },
    },
  };
</script>
